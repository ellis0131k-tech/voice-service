# Start Whisper STT only (port 8100)
cd whisper && source venv/Scripts/activate && uvicorn server:app --host 0.0.0.0 --port 8100

# Start Qwen3 TTS only (port 8200)
cd tts && source venv/Scripts/activate && uvicorn server:app --host 0.0.0.0 --port 8200

# Start both together (run from voice-service root)
(cd whisper && source venv/Scripts/activate && uvicorn server:app --host 0.0.0.0 --port 8100) & (cd tts && source venv/Scripts/activate && uvicorn server:app --host 0.0.0.0 --port 8200)
